A lightweight directive that is focused on providing async table with sorting, 
        pagination and filtering support.

        
Table Settings

        
Settings can be provided as attributes in the <tasty-table>.

        
    
          
  • 
            
    resource 
            : Current page number. First page is 1.
            
    
          
    • 
          
  • 
            
    filters 
            : Total number of items in all pages.
    
          
    •

$scope.getResource = function (params) {
  $scope.urlApi = 'table.json?' + params;
  return $http.get($scope.urlApi).then(function (response) {
    return {
      'rows': response.data.rows,
      'header': response.data.header,
      'pagination': response.data.pagination,
      'sortBy': response.data['sort-by'],
      'sortOrder': response.data['sort-order']
    }
  });
}
